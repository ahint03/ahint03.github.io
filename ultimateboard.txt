using System;
using System.Collections.Generic;
using System.Linq;
using System.Windows.Controls;
using System.Windows.Media;

namespace Project2
{
    public class UltimateBoard
    {
        private SmallBoard[,] boards = new SmallBoard[3, 3];
        private int nextRow = -1, nextCol = -1;
        private Grid bigGrid;
        private Action<object, System.Windows.RoutedEventArgs> clickHandler;
        public bool FreeMove { get; private set; } = false;

        public UltimateBoard(Grid bigGrid, Action<object, System.Windows.RoutedEventArgs> clickHandler)
        {
            this.bigGrid = bigGrid;
            this.clickHandler = clickHandler;
            InitializeBoards();
        }

        private void InitializeBoards()
        {
            bigGrid.Children.Clear();
            bigGrid.RowDefinitions.Clear();
            bigGrid.ColumnDefinitions.Clear();

            for (int i = 0; i < 3; i++)
            {
                bigGrid.RowDefinitions.Add(new RowDefinition());
                bigGrid.ColumnDefinitions.Add(new ColumnDefinition());
            }

            for (int row = 0; row < 3; row++)
            {
                for (int col = 0; col < 3; col++)
                {
                    Border border = new Border
                    {
                        BorderThickness = new System.Windows.Thickness(3),
                        BorderBrush = new SolidColorBrush((Color)ColorConverter.ConvertFromString("#D8A48F")),
                        Background = new SolidColorBrush((Color)ColorConverter.ConvertFromString("#FFF8F3")),
                        Margin = new System.Windows.Thickness(6),
                        CornerRadius = new System.Windows.CornerRadius(14)
                    };

                    Grid smallGrid = new Grid
                    {
                        Background = new SolidColorBrush((Color)ColorConverter.ConvertFromString("#FFF8F3"))
                    };

                    border.Child = smallGrid;
                    Grid.SetRow(border, row);
                    Grid.SetColumn(border, col);
                    bigGrid.Children.Add(border);

                    boards[row, col] = new SmallBoard(smallGrid)
                    {
                        Border = border
                    };

                    foreach (var btn in boards[row, col].Cells)
                        btn.Click += new System.Windows.RoutedEventHandler(clickHandler);
                }
            }

            HighlightActiveBoard();
        }

        public void ProcessMove(Button button, string symbol)
        {
            int bigRow = Grid.GetRow((button.Parent as Grid).Parent as Border);
            int bigCol = Grid.GetColumn((button.Parent as Grid).Parent as Border);

            int smallRow = Grid.GetRow(button);
            int smallCol = Grid.GetColumn(button);

            boards[bigRow, bigCol].CheckWin(symbol);

            nextRow = smallRow;
            nextCol = smallCol;

            FreeMove = false;
            if (nextRow >= 0 && nextCol >= 0)
            {
                if (boards[nextRow, nextCol].Winner != null || boards[nextRow, nextCol].IsFull())
                {
                    FreeMove = true;
                }
            }

            HighlightActiveBoard();
        }

        private void HighlightActiveBoard()
        {
            foreach (var small in boards)
            {
                small.Border.BorderThickness = new System.Windows.Thickness(3);
                small.Border.BorderBrush = new SolidColorBrush((Color)ColorConverter.ConvertFromString("#D8A48F"));
            }

            if (FreeMove)
            {
                foreach (var small in boards)
                {
                    if (small.Winner == null && !small.IsFull())
                    {
                        small.Border.BorderThickness = new System.Windows.Thickness(6);
                        small.Border.BorderBrush = Brushes.Gold;
                    }
                }
            }
            else if (nextRow >= 0 && nextCol >= 0 &&
                     boards[nextRow, nextCol] != null &&
                     boards[nextRow, nextCol].Winner == null &&
                     !boards[nextRow, nextCol].IsFull())
            {
                boards[nextRow, nextCol].Border.BorderThickness = new System.Windows.Thickness(6);
                boards[nextRow, nextCol].Border.BorderBrush = Brushes.Gold;
            }
        }

        public bool IsMoveValid(Button button)
        {
            int bigRow = Grid.GetRow((button.Parent as Grid).Parent as Border);
            int bigCol = Grid.GetColumn((button.Parent as Grid).Parent as Border);

            if (FreeMove || (nextRow == -1 && nextCol == -1))
                return true;

            return bigRow == nextRow && bigCol == nextCol;
        }

        public Button GetRandomMove()
        {
            var options = new List<Button>();

            if (!FreeMove && nextRow >= 0 && nextCol >= 0 &&
                boards[nextRow, nextCol] != null &&
                boards[nextRow, nextCol].Winner == null &&
                !boards[nextRow, nextCol].IsFull())
            {
                options.AddRange(boards[nextRow, nextCol].Cells.Cast<Button>().Where(b => b.Content == null));
            }
            else
            {
                foreach (var small in boards)
                {
                    if (small.Winner == null)
                        options.AddRange(small.Cells.Cast<Button>().Where(b => b.Content == null));
                }
            }

            if (options.Count == 0) return null;

            return options[new Random().Next(options.Count)];
        }

        public string GetBigWinner()
        {
            string[,] winners = new string[3, 3];

            for (int row = 0; row < 3; row++)
                for (int col = 0; col < 3; col++)
                    winners[row, col] = boards[row, col].Winner == "T" ? null : boards[row, col].Winner;

            for (int i = 0; i < 3; i++)
            {
                if (!string.IsNullOrEmpty(winners[i, 0]) &&
                    winners[i, 0] == winners[i, 1] &&
                    winners[i, 1] == winners[i, 2])
                    return winners[i, 0];

                if (!string.IsNullOrEmpty(winners[0, i]) &&
                    winners[0, i] == winners[1, i] &&
                    winners[1, i] == winners[2, i])
                    return winners[0, i];
            }

            if (!string.IsNullOrEmpty(winners[0, 0]) &&
                winners[0, 0] == winners[1, 1] &&
                winners[1, 1] == winners[2, 2])
                return winners[0, 0];

            if (!string.IsNullOrEmpty(winners[0, 2]) &&
                winners[0, 2] == winners[1, 1] &&
                winners[1, 1] == winners[2, 0])
                return winners[0, 2];

            return null;
        }

        public void DisableAllBoards()
        {
            foreach (var small in boards)
                small.DisableBoard();
        }
    }
}
