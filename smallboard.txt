using System.Linq;
using System.Windows;
using System.Windows.Controls;
using System.Windows.Media;

namespace Project2
{
    public class SmallBoard
    {
        public Button[,] Cells { get; private set; }
        public string Winner { get; private set; } = null;
        public Grid Container { get; private set; }
        public Border Border { get; set; }

        public SmallBoard(Grid container)
        {
            Container = container;
            InitializeBoard();
        }

        private void InitializeBoard()
        {
            Cells = new Button[3, 3];
            Container.Children.Clear();
            Container.RowDefinitions.Clear();
            Container.ColumnDefinitions.Clear();

            for (int i = 0; i < 3; i++)
            {
                Container.RowDefinitions.Add(new RowDefinition());
                Container.ColumnDefinitions.Add(new ColumnDefinition());
            }

            for (int row = 0; row < 3; row++)
            {
                for (int col = 0; col < 3; col++)
                {
                    Button btn = new Button
                    {
                        FontSize = 26,
                        FontWeight = FontWeights.Bold,
                        Margin = new Thickness(3),
                        Background = new SolidColorBrush((Color)ColorConverter.ConvertFromString("#FFFDF8")),
                        BorderBrush = new SolidColorBrush((Color)ColorConverter.ConvertFromString("#E7C6B4")),
                        BorderThickness = new Thickness(1.5),
                        Foreground = new SolidColorBrush((Color)ColorConverter.ConvertFromString("#5C3A36"))
                    };

                    Grid.SetRow(btn, row);
                    Grid.SetColumn(btn, col);
                    Container.Children.Add(btn);
                    Cells[row, col] = btn;
                }
            }
        }

        public bool CheckWin(string symbol)
        {
            for (int i = 0; i < 3; i++)
            {
                if (Cells[i, 0].Content?.ToString() == symbol &&
                    Cells[i, 1].Content?.ToString() == symbol &&
                    Cells[i, 2].Content?.ToString() == symbol)
                {
                    SetWinner(symbol);
                    return true;
                }

                if (Cells[0, i].Content?.ToString() == symbol &&
                    Cells[1, i].Content?.ToString() == symbol &&
                    Cells[2, i].Content?.ToString() == symbol)
                {
                    SetWinner(symbol);
                    return true;
                }
            }

            if (Cells[0, 0].Content?.ToString() == symbol &&
                Cells[1, 1].Content?.ToString() == symbol &&
                Cells[2, 2].Content?.ToString() == symbol)
            {
                SetWinner(symbol);
                return true;
            }

            if (Cells[0, 2].Content?.ToString() == symbol &&
                Cells[1, 1].Content?.ToString() == symbol &&
                Cells[2, 0].Content?.ToString() == symbol)
            {
                SetWinner(symbol);
                return true;
            }

            if (IsFull() && Winner == null)
            {
                SetTie();
            }

            return false;
        }

        private void SetWinner(string symbol)
        {
            Winner = symbol;

            Brush boardColor;
            Brush textColor;

            if (symbol == "X")
            {
                boardColor = new SolidColorBrush((Color)ColorConverter.ConvertFromString("#CBE9FF"));
                textColor = new SolidColorBrush((Color)ColorConverter.ConvertFromString("#007ACC"));
            }
            else
            {
                boardColor = new SolidColorBrush((Color)ColorConverter.ConvertFromString("#FFD6DC"));
                textColor = new SolidColorBrush((Color)ColorConverter.ConvertFromString("#E63946"));
            }

            Container.Background = boardColor;

            var bigText = new TextBlock
            {
                Text = symbol,
                FontSize = 120,
                FontWeight = FontWeights.Black,
                Foreground = textColor,
                HorizontalAlignment = HorizontalAlignment.Center,
                VerticalAlignment = VerticalAlignment.Center,
                TextAlignment = TextAlignment.Center,
                IsHitTestVisible = false
            };

            Grid.SetRowSpan(bigText, 3);
            Grid.SetColumnSpan(bigText, 3);

            Container.Children.Add(bigText);

            DisableBoard();
        }

        private void SetTie()
        {
            Winner = "T";
            Container.Background = new SolidColorBrush((Color)ColorConverter.ConvertFromString("#EADFD8"));

            var bigText = new TextBlock
            {
                Text = "T",
                FontSize = 110,
                FontWeight = FontWeights.ExtraBold,
                Foreground = Brushes.Gray,
                Opacity = 0.35,
                HorizontalAlignment = HorizontalAlignment.Center,
                VerticalAlignment = VerticalAlignment.Center,
                TextAlignment = TextAlignment.Center,
                IsHitTestVisible = false
            };

            Grid.SetRowSpan(bigText, 3);
            Grid.SetColumnSpan(bigText, 3);

            Container.Children.Add(bigText);
            DisableBoard();
        }

        public bool IsFull()
        {
            return Cells.Cast<Button>().All(b => b.Content != null);
        }

        public void DisableBoard()
        {
            foreach (var btn in Cells)
            {
                btn.IsEnabled = false;
            }
        }
    }
}
