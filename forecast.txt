    <div class="container">
        <h2>Weather Forecast</h2>
        <p class="location-name" id="locationName"></p>
        <div id="weatherResult"></div>
        <button onclick="toggleTemperatureUnit()">Toggle Celsius/Fahrenheit</button>
        <button onclick="goBack()">Go Back</button>
    </div>

    <script>
        let isCelsius = true;
        let forecastDays = [];

        function celsiusToFahrenheit(temp) {
            return Math.round((temp * 9 / 5) + 32);
        }

        function formatTemperature(tempC) {
            return isCelsius
                ? `${Math.round(tempC)} °C`
                : `${celsiusToFahrenheit(tempC)} °F`;
        }

        async function getCoordinates(location) {
            const apiKey = '1d7fdf2b85274470ae8373de4da0a89e';
            const searchUrl = `https://api.opencagedata.com/geocode/v1/json?q=${encodeURIComponent(location)}&key=${apiKey}`;

            try {
                const response = await fetch(searchUrl);
                if (!response.ok) {
                    throw new Error(`Failed to fetch coordinates: ${response.statusText}`);
                }

                const data = await response.json();

                if (data.results && data.results.length > 0) {
                    const { lat, lng } = data.results[0].geometry;
                    reverseGeocode(lat, lng);
                    getWeather(lat, lng);
                } else {
                    throw new Error('Location not found');
                }
            } catch (error) {
                console.error('Error fetching coordinates:', error);
                document.getElementById('weatherResult').innerText =
                    'Unable to fetch weather data. Please make sure the location is correct.';
                document.getElementById('locationName').innerText = location;
            }
        }

        async function reverseGeocode(lat, lng) {
            const apiKey = '1d7fdf2b85274470ae8373de4da0a89e';
            const reverseGeocodeUrl = `https://api.opencagedata.com/geocode/v1/json?q=${lat}+${lng}&key=${apiKey}`;

            try {
                const response = await fetch(reverseGeocodeUrl);
                if (!response.ok) {
                    throw new Error(`Failed to reverse geocode coordinates: ${response.statusText}`);
                }

                const data = await response.json();

                if (data.results && data.results.length > 0) {
                    const components = data.results[0].components;
                    const city = components.city || components.town || components.village || 'Unknown city';
                    const state = components.state || 'Unknown state';
                    document.getElementById('locationName').innerText = `${city}, ${state}`;
                } else {
                    document.getElementById('locationName').innerText = 'Unknown location';
                }
            } catch (error) {
                console.error('Error reverse geocoding:', error);
                document.getElementById('locationName').innerText = 'Unable to fetch location name';
            }
        }

        async function getWeather(lat, lng) {
            const apiKey = 'YPe1BIuAOcdzHWCbeuZGzN7xRGAQRV9u';
            const weatherUrl = `https://api.tomorrow.io/v4/weather/forecast?location=${lat},${lng}&apikey=${apiKey}`;

            try {
                const response = await fetch(weatherUrl);

                if (!response.ok) {
                    throw new Error(`Weather data error: ${response.status} ${response.statusText}`);
                }

                const data = await response.json();
                console.log("Tomorrow forecast data:", data);
                displayWeather(data);
            } catch (error) {
                console.error('Error fetching weather data:', error);
                document.getElementById('weatherResult').innerText =
                    'Unable to fetch weather data. The weather service may be unavailable or the API key may be invalid.';
            }
        }

        function mapWeatherCodeToDescription(weatherCode) {
            switch (weatherCode) {
                case 1000: return 'Clear';
                case 1001: return 'Cloudy';
                case 1100: return 'Mostly Clear';
                case 1101: return 'Partly Cloudy';
                case 1102: return 'Mostly Cloudy';
                case 2000: return 'Fog';
                case 2100: return 'Light Fog';
                case 4000: return 'Drizzle';
                case 4001: return 'Rain';
                case 4200: return 'Light Rain';
                case 4201: return 'Heavy Rain';
                case 5000: return 'Snow';
                case 5001: return 'Flurries';
                case 5100: return 'Light Snow';
                case 5101: return 'Heavy Snow';
                case 6000: return 'Freezing Drizzle';
                case 6001: return 'Freezing Rain';
                case 6200: return 'Light Freezing Rain';
                case 6201: return 'Heavy Freezing Rain';
                case 7000: return 'Ice Pellets';
                case 7101: return 'Heavy Ice Pellets';
                case 7102: return 'Light Ice Pellets';
                case 8000: return 'Thunderstorm';
                default: return 'Unknown';
            }
        }

        function getWeatherImage(weatherCode) {
            const weatherImages = {
                1000: 'sun.png',
                1001: 'cloud.png',
                1100: 'sun.png',
                1101: 'partly.png',
                1102: 'cloud.png',
                2000: 'cloud.png',
                2100: 'cloud.png',
                4000: 'rain.png',
                4001: 'rain.png',
                4200: 'rain.png',
                4201: 'rain.png',
                5000: 'snow.png',
                5001: 'snow.png',
                5100: 'snow.png',
                5101: 'snow.png',
                6000: 'rain.png',
                6001: 'rain.png',
                6200: 'rain.png',
                6201: 'rain.png',
                7000: 'snow.png',
                7101: 'snow.png',
                7102: 'snow.png',
                8000: 'thunder.png'
            };

            return weatherImages[weatherCode] || 'default.png';
        }

        function displayWeather(data) {
            const weatherResult = document.getElementById('weatherResult');

            if (data && data.timelines && data.timelines.daily && data.timelines.daily.length > 0) {
                forecastDays = data.timelines.daily.slice(0, 5);

                let html = `<h3>5-Day Weather Forecast</h3><div class="forecast-row">`;

                forecastDays.forEach(day => {
                    const date = new Date(day.time).toLocaleDateString();
                    const temperatureMax = day.values.temperatureMax;
                    const temperatureMin = day.values.temperatureMin;
                    const weatherCode = day.values.weatherCodeMax ?? day.values.weatherCodeMin ?? day.values.weatherCode;
                    const weatherDescription = mapWeatherCodeToDescription(weatherCode);
                    const weatherImage = getWeatherImage(weatherCode);

                    html += `
                        <div class="weather-block">
                            <h3>${date}</h3>
                            <p>Max Temperature: ${formatTemperature(temperatureMax)}</p>
                            <p>Min Temperature: ${formatTemperature(temperatureMin)}</p>
                            <p>Weather: ${weatherDescription}</p>
                            <img src="${weatherImage}" alt="${weatherDescription}">
                        </div>
                    `;
                });

                html += `</div>`;
                weatherResult.innerHTML = html;
            } else {
                weatherResult.innerHTML = '<p>No weather data available for this location.</p>';
            }
        }

        function toggleTemperatureUnit() {
            isCelsius = !isCelsius;

            if (forecastDays.length > 0) {
                displayWeather({
                    timelines: {
                        daily: forecastDays
                    }
                });
            }
        }

        function goBack() {
            window.location.href = 'index.html';
        }

        const urlParams = new URLSearchParams(window.location.search);
        const locationParam = urlParams.get('location');

        if (locationParam) {
            getCoordinates(locationParam);
        }
